## 정수 2개를 입력받아 두 정수의 합을 구하는 c언어 코드
```c

#include <stdio.h>
  int main()
  {
    int a,b;
    int sum=0;
    printf("정수 a, b를 입력하시오: ");
    scanf("%d %d", &a, &b);
    sum=a+b;
    printf("두 수의 합은 %d이다.", sum);
    return 0;
  }
  ```
